116  {
117  CountType i;
118  for (i=count; i>0; --i) {
119 #ifdef NANOFLANN_FIRST_MATCH // If defined and two points have the same distance, the one with the lowest-index will be returned first.

580  only works for WORDSIZE that is a power of 2, by masking last bits of
581  incremented size to zero.
582  */
583  const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
584 
585  /* Check whether a new block must be allocated. Note that the first word
586  of a block is reserved for a pointer to the previous block.
587  */
588  if (size > remaining) {
589 
590  wastedMemory += remaining;
591 
592  /* Allocate new storage. */
593  const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
594  size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
595 
596  // use the standard C malloc to allocate memory
597  void* m = ::malloc(blocksize);
598  if (!m) {
599  fprintf(stderr,"Failed to allocate memory.\n");
600  return NULL;
601  }
602 
603  /* Fill first word of new block with pointer to previous block. */
604  static_cast<void**>(m)[0] = base;
605  base = m;
606 
607  size_t shift = 0;
608  //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
609 
610  remaining = blocksize - sizeof(void*) - shift;
611  loc = (static_cast<char*>(m) + sizeof(void*) + shift);
612  }
613  void* rloc = loc;
614  loc = static_cast<char*>(loc) + size;
615  remaining -= size;
616

857  NodePtr divideTree(Derived &obj, const IndexType left, const IndexType right, BoundingBox& bbox)
858  {
859  NodePtr node = obj.pool.allocate<Node>(); // allocate memory
860 
861  /* If too few exemplars remain, then make this a leaf node. */
862  if ( (right-left) <= static_cast<IndexType>(obj.m_leaf_max_size) ) {
863  node->child1 = node->child2 = NULL; /* Mark as leaf node. */
864  node->node_type.lr.left = left;

866

868  for (int i=0; i<(DIM>0 ? DIM : obj.dim); ++i) {
869  bbox[i].low = obj.dataset_get(obj.vind[left],i);
870  bbox[i].high = obj.dataset_get(obj.vind[left],i);
871  }
872  for (IndexType k=left+1; k<right; ++k) {
873  for (int i=0; i<(DIM>0 ? DIM : obj.dim); ++i) {
874  if (bbox[i].low>obj.dataset_get(obj.vind[k],i)) bbox[i].low=obj.dataset_get(obj.vind[k],i);
875  if (bbox[i].high<obj.dataset_get(obj.vind[k],i)) bbox[i].high=obj.dataset_get(obj.vind[k],i);
876  }
877  }
878  }
879  else {
880  IndexType idx;
881  int cutfeat;
882  DistanceType cutval;
883  middleSplit_(obj, &obj.vind[0]+left, right-left, idx, cutfeat, cutval, bbox);
884 
885  node->node_type.sub.divfeat = cutfeat;
886 
887  BoundingBox left_bbox(bbox);
888  left_bbox[cutfeat].high = cutval;
889  node->child1 = divideTree(obj, left, left+idx, left_bbox);
890 
891  BoundingBox right_bbox(bbox);
892  right_bbox[cutfeat].low = cutval;
893  node->child2 = divideTree(obj, left+idx, right, right_bbox);
894 
895  node->node_type.sub.divlow = left_bbox[cutfeat].high;
896  node->node_type.sub.divhigh = right_bbox[cutfeat].low;
897 
898  for (int i=0; i<(DIM>0 ? DIM : obj.dim); ++i) {
899  bbox[i].low = std::min(left_bbox[i].low, right_bbox[i].low);
900  bbox[i].high = std::max(left_bbox[i].high, right_bbox[i].high);
901  }
902  }
903 
904  return node;
905  }

907  void middleSplit_(Derived &obj, IndexType* ind, IndexType count, IndexType& index, int& cutfeat, DistanceType& cutval, const BoundingBox& bbox)
908  {
909  const DistanceType EPS=static_cast<DistanceType>(0.00001);
910  ElementType max_span = bbox[0].high-bbox[0].low;
911  for (int i=1; i<(DIM>0 ? DIM : obj.dim); ++i) {
912  ElementType span = bbox[i].high-bbox[i].low;
913  if (span>max_span) {
914  max_span = span;
915  }
916  }
917  ElementType max_spread = -1;
918  cutfeat = 0;
919  for (int i=0; i<(DIM>0 ? DIM : obj.dim); ++i) {
920  ElementType span = bbox[i].high-bbox[i].low;
921  if (span>(1-EPS)*max_span) {
922  ElementType min_elem, max_elem;
923  computeMinMax(obj, ind, count, i, min_elem, max_elem);
924  ElementType spread = max_elem-min_elem;;
925  if (spread>max_spread) {
926  cutfeat = i;
927  max_spread = spread;
928  }
929  }
930  }
931  // split in the middle
932  DistanceType split_val = (bbox[cutfeat].low+bbox[cutfeat].high)/2;
933  ElementType min_elem, max_elem;
934  computeMinMax(obj, ind, count, cutfeat, min_elem, max_elem);
935 
936  if (split_val<min_elem) cutval = min_elem;
937  else if (split_val>max_elem) cutval = max_elem;
938  else cutval = split_val;
939 
940  IndexType lim1, lim2;
941  planeSplit(obj, ind, count, cutfeat, cutval, lim1, lim2);
942 
943  if (lim1>count/2) index = lim1;
944  else if (lim2<count/2) index = lim2;
945  else index = count/2;
946  }

957  void planeSplit(Derived &obj, IndexType* ind, const IndexType count, int cutfeat, DistanceType &cutval, IndexType& lim1, IndexType& lim2)
958  {
959  /* Move vector indices for left subtree to front of list. */
960  IndexType left = 0;
961  IndexType right = count-1;
962  for (;; ) {
963  while (left<=right && obj.dataset_get(ind[left],cutfeat)<cutval) ++left;
964  while (right && left<=right && obj.dataset_get(ind[right],cutfeat)>=cutval) --right;
965  if (left>right || !right) break; // "!right" was added to support unsigned Index types
966  std::swap(ind[left], ind[right]);
967  ++left;
968  --right;
969  }
970  /* If either list is empty, it means that all remaining features
971  * are identical. Split in the middle to maintain a balanced tree.
972  */
973  lim1 = left;
974  right = count-1;
975  for (;; ) {
976  while (left<=right && obj.dataset_get(ind[left],cutfeat)<=cutval) ++left;
977  while (right && left<=right && obj.dataset_get(ind[right],cutfeat)>cutval) --right;
978  if (left>right || !right) break; // "!right" was added to support unsigned Index types
979  std::swap(ind[left], ind[right]);
980  ++left;
981  --right;
982  }
983  lim2 = left;
984  }

1296  template <class RESULTSET>
1297  bool searchLevel(RESULTSET& result_set, const ElementType* vec, const NodePtr node, DistanceType mindistsq,
1298  distance_vector_t& dists, const float epsError) const
1299  {
1300  /* If this is a leaf node, then do check and return. */
1301  if ((node->child1 == NULL)&&(node->child2 == NULL)) {
1302  //count_leaf += (node->lr.right-node->lr.left); // Removed since was neither used nor returned to the user.
1303  DistanceType worst_dist = result_set.worstDist();
1304  for (IndexType i=node->node_type.lr.left; i<node->node_type.lr.right; ++i) {
1305  const IndexType index = vind[i];// reorder... : i;
1306  DistanceType dist = distance(vec, index, (DIM>0 ? DIM : dim));
1307  if (dist<worst_dist) {
1308  if(!result_set.addPoint(dist,vind[i])) {
1309  // the resultset doesn't want to receive any more points, we're done searching!
1310  return false;
1311  }
1312  }
1313  }
1314  return true;
1315  }
1316 
1317  /* Which child branch should be taken first? */
1318  int idx = node->node_type.sub.divfeat;
1319  ElementType val = vec[idx];
1320  DistanceType diff1 = val - node->node_type.sub.divlow;
1321  DistanceType diff2 = val - node->node_type.sub.divhigh;
1322 
1323  NodePtr bestChild;
1324  NodePtr otherChild;
1325  DistanceType cut_dist;
1326  if ((diff1+diff2)<0) {
1327  bestChild = node->child1;
1328  otherChild = node->child2;
1329  cut_dist = distance.accum_dist(val, node->node_type.sub.divhigh, idx);
1330  }
1331  else {
1332  bestChild = node->child2;
1333  otherChild = node->child1;
1334  cut_dist = distance.accum_dist( val, node->node_type.sub.divlow, idx);
1335  }
1336 
1337  /* Call recursively to search next level down. */
1338  if(!searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError)) {
1339  // the resultset doesn't want to receive any more points, we're done searching!
1340  return false;
1341  }
1342 
1343  DistanceType dst = dists[idx];
1344  mindistsq = mindistsq + cut_dist - dst;
1345  dists[idx] = cut_dist;
1346  if (mindistsq*epsError<=result_set.worstDist()) {
1347  if(!searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError)) {
1348  // the resultset doesn't want to receive any more points, we're done searching!
1349  return false;
1350  }
1351  }
1352  dists[idx] = dst;
1353  return true;
1354  }
